def get_tags(names)
  if names.any?
    names
  else
    names << %w(tbt cute happy fashion followme me follow like4like selfie winter picoftheday summer friends instadaily girl fun repost love instagood photooftheday beautiful )
    names.flatten!
  end
  names.map { |name| Tag.new(self, name) }
end
